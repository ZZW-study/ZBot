```python
class AgentLoop:
    def __init__(
        self,
```

### 核心方法详解

#### 方法1：`run()` - 主循环入口

```python
async def run(self) -> None:
    """
    启动Agent主循环：持续监听消息总线，处理入站消息
    """
    self._running = True
    
    while self._running:
        try:
            # 1. 从消息总线获取入站消息（阻塞等待）
            msg = await self.bus.consume_inbound()
            
            # 2. 处理消息
            response = await self.process_direct(
                message=msg.content,
                session_key=f"{msg.channel}:{msg.chat_id}",
                channel=msg.channel,
                chat_id=msg.chat_id,
            )
            
            # 3. 发布出站消息
            await self.bus.publish_outbound(OutboundMessage(
                channel=msg.channel,
                chat_id=msg.chat_id,
                content=response,
            ))
            
        except asyncio.CancelledError:
            break
        except Exception as e:
            logger.exception("Agent loop error: {}", e)
```

**关键点**：
- 使用异步循环持续监听
- 入站/出站消息分离处理
- 异常不中断循环，只记录日志

#### 方法2：`process_direct()` - 单次处理

```python
async def process_direct(
    self,
    message: str,
    session_key: str = "cli:direct",
    channel: str = "cli",
    chat_id: str = "direct",
    on_progress: Callable | None = None,
) -> str:
    """
    直接处理单条消息（核心推理逻辑）
    """
    # 1. 获取/创建会话
    session = self.session_manager.get_or_create(session_key)
    
    # 2. 构建消息上下文
    history = session.get_history(self.memory_window)
    messages = self.context.build_messages(
        history=history,
        current_message=message,
        channel=channel,
        chat_id=chat_id,
    )
    
    # 3. 迭代调用LLM（处理工具调用）
    iterations = 0
    while iterations < self.max_iterations:
        # 调用LLM
        response = await self.provider.chat(
            messages=messages,
            tools=self.tools.to_schemas(),
            model=self.model,
            max_tokens=self.max_tokens,
            temperature=self.temperature,
        )
        
        # 无工具调用，返回结果
        if not response.tool_calls:
            break
        
        # 有工具调用，执行工具
        iterations += 1
        
        # 进度回调
        if on_progress:
            await on_progress(f"Executing tools...", tool_hint=True)
        
        # 执行所有工具调用
        for tc in response.tool_calls:
            result = await self._execute_tool(tc, on_progress)
            
            # 添加工具结果到消息
            messages = self.context.add_tool_result(
                messages, tc.id, tc.name, result
            )
    
    # 4. 保存会话
    session.add_message("user", message)
    session.add_message("assistant", response.content or "")
    self.session_manager.save(session)
    
    return response.content or ""
```

**关键点**：
- 工具调用迭代：最多 `max_iterations` 次
- 消息历史累积：每次工具结果都加入上下文
- 会话持久化：处理完成后立即保存

#### 方法3：`_execute_tool()` - 工具执行

```python
async def _execute_tool(
    self,
    tool_call: ToolCallRequest,
    on_progress: Callable | None = None,
) -> str:
    """
    执行单个工具调用
    """
    # 1. 获取工具实例
    tool = self.tools.get(tool_call.name)
    if not tool:
        return f"Error: Unknown tool '{tool_call.name}'"
    
    # 2. 类型转换（根据工具参数Schema）
    params = tool.cast_params(tool_call.arguments)
    
    # 3. 参数校验
    errors = tool.validate_params(params)
    if errors:
        return f"Error: Invalid parameters: {', '.join(errors)}"
    
    # 4. 进度回调
    if on_progress:
        await on_progress(f"Calling {tool_call.name}...")
    
    # 5. 执行工具
    try:
        result = await tool.execute(**params)
        return result
    except Exception as e:
        return f"Error: {str(e)}"
```

```python
class LiteLLMProvider(LLMProvider):
    def __init__(
        self,
        api_key: str | None = None,
        api_base: str | None = None,
        default_model: str = "anthropic/claude-opus-4-5",
        extra_headers: dict[str, str] | None = None,
        provider_name: str | None = None,
    ):
        self.default_model = default_model
        self.extra_headers = extra_headers or {}
        
        # 自动检测网关/本地部署
        self._gateway = find_gateway(provider_name, api_key, api_base)
        
        # LiteLLM配置
        litellm.suppress_debug_info = True
        litellm.drop_params = True  # 自动删除不支持的参数
```

### 核心方法详解

#### 方法1：`chat()` - 核心调用

```python
async def chat(
    self,
    messages: list[dict[str, Any]],
    tools: list[dict[str, Any]] | None = None,
    model: str | None = None,
    max_tokens: int = 4096,
    temperature: float = 0.7,
    reasoning_effort: str | None = None,
) -> LLMResponse:
    """
    异步发送聊天请求 - 完整流程
    """
    # 1. 模型名称标准化
    original_model = model or self.default_model
    model = self._resolve_model(original_model)
    
    # 2. 提示词缓存注入
    if self._supports_cache_control(original_model):
        messages, tools = self._apply_cache_control(messages, tools)
    
    # 3. 构建请求参数
    kwargs: dict[str, Any] = {
        "model": model,
        "messages": self._sanitize_messages(messages),
        "max_tokens": max(1, max_tokens),
        "temperature": temperature,
    }
    
    # 4. 工具调用配置
    if tools:
        kwargs["tools"] = tools
        kwargs["tool_choice"] = "auto"
    
    # 5. 调用LiteLLM
    response = await acompletion(**kwargs)
    
    # 6. 解析响应
    return self._parse_response(response)
```

#### 方法2：`_resolve_model()` - 模型名称标准化

```python
def _resolve_model(self, model: str) -> str:
    """
    模型名称标准化：
    - 网关模式：剥离原有前缀 → 添加网关前缀
    - 标准模式：自动添加厂商前缀
    """
    if self._gateway:
        # 网关模式
        prefix = self._gateway.litellm_prefix
        if self._gateway.strip_model_prefix:
            model = model.split("/")[-1]
        if prefix and not model.startswith(f"{prefix}/"):
            model = f"{prefix}/{model}"
        return model
    
    # 标准厂商：自动添加前缀
    spec = find_by_model(model)
    if spec and spec.litellm_prefix:
        if not any(model.startswith(s) for s in spec.skip_prefixes):
            model = f"{spec.litellm_prefix}/{model}"
    
    return model
```

**示例**：
- `"claude-3-opus"` → `"anthropic/claude-3-opus"`
- `"gpt-4o"` → `"openai/gpt-4o"`
- `"deepseek-chat"` → `"deepseek/deepseek-chat"`

#### 方法3：`_sanitize_messages()` - 消息清洗

```python
@staticmethod
def _sanitize_messages(
    messages: list[dict[str, Any]],
    extra_keys: frozenset[str] = frozenset(),
) -> list[dict[str, Any]]:
    """
    消息最终清洗：
    1. 保留标准字段 + 厂商专属字段
    2. 标准化工具调用ID（兼容所有厂商）
    """
    allowed = _ALLOWED_MSG_KEYS | extra_keys
    sanitized = []
    id_map: dict[str, str] = {}
    
    for msg in messages:
        # 过滤非法字段
        clean = {k: v for k, v in msg.items() if k in allowed}
        
        # 标准化工具调用ID
        if "tool_calls" in clean:
            for tc in clean["tool_calls"]:
                tc["id"] = id_map.setdefault(
                    tc["id"],
                    _normalize_tool_id(tc["id"])
                )
        
        sanitized.append(clean)
    
    return sanitized
```

#### 方法1：`build_system_prompt()` - 构建系统提示

```python
def build_system_prompt(self, skill_names: list[str] | None = None) -> str:
    """
    构建AI的系统提示词（身份、规则、能力、记忆）
    """
    parts = []
    
    # 1. AI身份信息（我是谁、运行环境、规则）
    parts.append(self._get_identity())
    
    # 2. 引导文件（AGENTS.md等自定义规则）
    bootstrap = self._load_bootstrap_files()
    if bootstrap:
        parts.append(bootstrap)
    
    # 3. 长期记忆（MEMORY.md）
    memory = self.memory.get_memory_context()
    if memory:
        parts.append(f"# Memory\n\n{memory}")
    
    # 4. 常驻技能（always=true的技能）
    always_skills = self.skills.get_always_skills()
    if always_skills:
        always_content = self.skills.load_skills_for_context(always_skills)
        if always_content:
            parts.append(f"# Active Skills\n\n{always_content}")
    
    # 5. 技能摘要（XML格式的技能清单）
    skills_summary = self.skills.build_skills_summary()
    if skills_summary:
        parts.append(f"# Skills\n{skills_summary}")
    
    # 用分隔线拼接
    return "\n\n---\n\n".join(parts)
```

#### 方法2：`build_messages()` - 构建完整消息

```python
def build_messages(
    self,
    history: list[dict[str, Any]],
    current_message: str,
    skill_names: list[str] | None = None,
    media: list[str] | None = None,
    channel: str | None = None,
    chat_id: str | None = None,
) -> list[dict[str, Any]]:
    """
    构建AI大模型需要的完整消息列表
    格式：[系统提示, 历史消息1, ..., 当前用户消息]
    """
    # 1. 运行时上下文（时间、渠道）
    runtime_ctx = self._build_runtime_context(channel, chat_id)
    
    # 2. 处理用户消息（支持图片）
    user_content = self._build_user_content(current_message, media)
    
    # 3. 合并运行时上下文和用户消息
    if isinstance(user_content, str):
        merged = f"{runtime_ctx}\n\n{user_content}"
    else:
        # 多模态消息：分开组装
        merged = [{"type": "text", "text": runtime_ctx}] + user_content
    
    # 4. 返回最终消息列表
    return [
        {"role": "system", "content": self.build_system_prompt(skill_names)},
        *history,
        {"role": "user", "content": merged},
    ]
```

```python
def _get_identity(self) -> str:
    """

    workspace_path = str(self.workspace.expanduser().resolve())

    return f"""# nanobot 🐈
You are nanobot, a helpful AI assistant.

## Runtime
{runtime}

## Workspace
Your workspace is at: {workspace_path}
- Long-term memory: {workspace_path}/memory/MEMORY.md
- History log: {workspace_path}/memory/HISTORY.md
- Custom skills: {workspace_path}/skills/{{skill-name}}/SKILL.md

## nanobot Guidelines

- Before modifying a file, read it first.
- If a tool call fails, analyze the error before retrying.
- Ask for clarification when the request is ambiguous.
"""
```

## 关键设计模式总结

### 1. AgentLoop：模板方法模式
```
run() → process_direct() → _execute_tool()
  │           │                  │
  │           │                  └── 具体工具执行
  │           └── LLM调用 + 工具迭代
  └── 消息循环框架
```

### 2. LiteLLMProvider：策略模式
```
ProviderSpec（策略对象）驱动不同厂商的特殊处理
- litellm_prefix: 模型前缀
- supports_prompt_caching: 是否支持缓存
- model_overrides: 模型专属参数
```

### 3. ContextBuilder：建造者模式
```
build_messages() 分步组装：
  身份信息 → 引导文件 → 记忆 → 技能 → 运行时上下文 → 用户消息
```
