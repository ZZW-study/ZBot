**核心方法1：消息处理入口**
```python
async def process_direct(
    self,
    message: str,
    session_key: str = "cli:direct",
    channel: str = "cli",
    chat_id: str = "direct",
    on_progress: Callable | None = None,
) -> str:
    """
    直接处理单条消息（不走消息总线）
    用于CLI单次对话和定时任务执行
    """
    # 1. 获取/创建会话
    session = self.session_manager.get_or_create(session_key)
    
    # 2. 构建完整消息上下文
    history = session.get_history(self.memory_window)
    messages = self.context.build_messages(
        history=history,
        current_message=message,
        channel=channel,
        chat_id=chat_id,
    )
    
    # 3. 调用LLM
    response = await self.provider.chat(
        messages=messages,
        tools=self.tools.to_schemas(),
        model=self.model,
        ...
    )
    
    # 4. 处理工具调用
    while response.tool_calls:
        # 执行工具
        tool_results = await self._execute_tool_calls(response.tool_calls)
        # 更新消息历史
        messages = self.context.add_tool_result(messages, tool_results)
        # 再次调用LLM
        response = await self.provider.chat(messages=messages, ...)
    
    # 5. 保存会话并返回
    session.add_message("assistant", response.content)
    self.session_manager.save(session)
    
    return response.content
```

**核心方法2：工具执行**
```python
async def _execute_tool_calls(self, tool_calls: list[ToolCallRequest]) -> str:
    """执行工具调用并返回结果"""
    results = []
    for tc in tool_calls:
        # 获取工具实例
        tool = self.tools.get(tc.name)
        if not tool:
            results.append(f"Error: Unknown tool {tc.name}")
            continue
        
        # 执行工具
        try:
            result = await tool.execute(**tc.arguments)
            results.append(f"[{tc.name}] {result}")
        except Exception as e:
            results.append(f"[{tc.name}] Error: {str(e)}")
    
    return "\n".join(results)
```

**核心方法：chat()**
```python
async def chat(
    self,
    messages: list[dict[str, Any]],
    tools: list[dict[str, Any]] | None = None,
    model: str | None = None,
    max_tokens: int = 4096,
    temperature: float = 0.7,
    reasoning_effort: str | None = None,
) -> LLMResponse:
    """
    核心方法：异步发送聊天请求
    
    完整流程：
    1. 模型名称解析（添加/剥离前缀）
    2. 提示词缓存注入（Anthropic/OpenRouter）
    3. 消息清洗（过滤非法字段）
    4. 参数覆盖（如Kimi强制温度1.0）
    5. 调用LiteLLM
    6. 响应解析（合并多Choice）
    """
    # 1. 模型名称标准化
    original_model = model or self.default_model
    model = self._resolve_model(original_model)
    
    # 2. 提示词缓存注入
    if self._supports_cache_control(original_model):
        messages, tools = self._apply_cache_control(messages, tools)
    
    # 3. 构建请求参数
    kwargs = {
        "model": model,
        "messages": self._sanitize_messages(messages),
        "max_tokens": max(1, max_tokens),
        "temperature": temperature,
    }
    
    # 4. 工具调用配置
    if tools:
        kwargs["tools"] = tools
        kwargs["tool_choice"] = "auto"
    
    # 5. 异步调用LiteLLM
    response = await acompletion(**kwargs)
    
    # 6. 解析响应
    return self._parse_response(response)
```

**关键设计：模型名称标准化**
```python
def _resolve_model(self, model: str) -> str:
    """
    模型名称标准化：
    - 网关模式：剥离原有前缀 → 添加网关前缀
    - 标准模式：应用厂商前缀
    
    示例：
    - "claude-3-opus" → "anthropic/claude-3-opus"
    - "gpt-4o" → "openai/gpt-4o"
    """
    if self._gateway:
        prefix = self._gateway.litellm_prefix
        if self._gateway.strip_model_prefix:
            model = model.split("/")[-1]
        if prefix and not model.startswith(f"{prefix}/"):
            model = f"{prefix}/{model}"
        return model
    
    # 标准厂商：自动添加前缀
    spec = find_by_model(model)
    if spec and spec.litellm_prefix:
        if not any(model.startswith(s) for s in spec.skip_prefixes):
            model = f"{spec.litellm_prefix}/{model}"
    
    return model
```

**核心方法：build_system_prompt()**
```python
def build_system_prompt(self, skill_names: list[str] | None = None) -> str:
    """
    构建AI的系统提示词（AI的身份、规则、能力、记忆）
    """
    parts = []
    
    # 1. AI身份信息
    parts.append(self._get_identity())
    
    # 2. 引导文件（AGENTS.md等）
    bootstrap = self._load_bootstrap_files()
    if bootstrap:
        parts.append(bootstrap)
    
    # 3. 长期记忆
    memory = self.memory.get_memory_context()
    if memory:
        parts.append(f"# Memory\n\n{memory}")
    
    # 4. 常驻技能
    always_skills = self.skills.get_always_skills()
    if always_skills:
        always_content = self.skills.load_skills_for_context(always_skills)
        if always_content:
            parts.append(f"# Active Skills\n\n{always_content}")
    
    # 5. 技能摘要
    skills_summary = self.skills.build_skills_summary()
    if skills_summary:
        parts.append(f"# Skills\n{skills_summary}")
    
    # 用分隔符拼接
    return "\n\n---\n\n".join(parts)
```

**核心方法：build_messages()**
```python
def build_messages(
    self,
    history: list[dict[str, Any]],
    current_message: str,
    skill_names: list[str] | None = None,
    media: list[str] | None = None,
    channel: str | None = None,
    chat_id: str | None = None,
) -> list[dict[str, Any]]:
    """
    构建AI大模型需要的完整消息列表
    格式：[系统提示, 历史消息1, ..., 当前用户消息]
    """
    # 1. 运行时上下文（时间、渠道）
    runtime_ctx = self._build_runtime_context(channel, chat_id)
    
    # 2. 处理用户消息（文本+图片）
    user_content = self._build_user_content(current_message, media)
    
    # 3. 合并运行时上下文和用户消息
    if isinstance(user_content, str):
        merged = f"{runtime_ctx}\n\n{user_content}"
    else:
        merged = [{"type": "text", "text": runtime_ctx}] + user_content
    
    # 4. 返回完整消息列表
    return [
        {"role": "system", "content": self.build_system_prompt(skill_names)},
        *history,
        {"role": "user", "content": merged},
    ]
```
